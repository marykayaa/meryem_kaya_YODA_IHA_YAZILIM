class Kedi:
    # Kurucu metot: Kedi oluşturulduğunda adı ve rengi belirlenir.
    def __init__(self, isim, renk):
        self.isim = isim # Özellik
        self.renk = renk # Özellik
        self.ac = True  # Varsayılan özellik

       
Nesne Yaratma "pamuk = Kedi(""Pamuk"", ""Beyaz"")" Kedi sınıfını çağırarak bellekte yeni bir Kedi nesnesi (pamuk) oluşturulur.
